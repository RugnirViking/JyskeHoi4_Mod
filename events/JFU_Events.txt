	id = jfu.11
	title = jfu.11.t
	desc = jfu.11.d
	picture = GFX_report_event_polish_tanks_01

	trigger = {
		AND = {
			100 = { is_owned_by = JFU } # Iceland
			101 = { is_owned_by = JFU } # Greenland

		}
	}

		name = jfu.11.a
		ISN = {
			transfer_state = 100
			transfer_state = 101
			transfer_state = 337
		}
		100 = {
			add_core_of = ISN
		}
		101 = {
			add_core_of = ISN
		}
		337 = {
			add_core_of = ISN
		}
	}
	option = { # create three separate states

			transfer_state = 100
		}
		100 = {
			add_core_of = ISI
		}
		ISF = {
			transfer_state = 337
		}
		337 = {
			add_core_of = ISF
		}
		ISG = {
			transfer_state = 101
		}

			add_core_of = ISG
		}
	}
